  \question

  Para una función arbitraria $f\colon X\rightarrow Y$, pruebe que
  las siguientes relaciones se mantienen:

  \begin{parts}
    \part

    \begin{math}
      f
      \left(\bigcup_{i\in I}A_{i}\right)=
      \bigcup_{i\in I}f\left(A_{i}\right).
    \end{math}

    \part

    \begin{math}
      f
      \left(\bigcap_{i\in I}A_{i}\right)\subseteq
      \bigcap_{i\in I}f\left(A_{i}\right).
    \end{math}

    \part

    Brinde un contraejemplo en el ítem anterior para que no se
    compruebe la igualdad.

  \end{parts}

  \begin{solution}
    \begin{parts}
      \part

      Sea $y\in f\left(\bigcup_{i\in I}A_{i}\right)$.

      Por definición, esto significa que
      \begin{math}
        \exists\,x\in
        \left(\bigcup_{i\in I}A_{i}\right)\ni
        f\left(x\right)=
        y.
      \end{math}

      Esto es,
      \begin{math}
        \exists\,x\ni
        \exists\,i\in
        I\ni
        x\in
        A_{i}
      \end{math}
      y $f\left(x\right)=y$.

      Así, $\exists\,i\in I\ni y\in f\left(A_{i}\right)$.

      Por lo tanto, $y\in\bigcup_{i\in I}f\left(A_{i}\right)$.

      \part

      Sea $y\in f\left(\bigcap_{i\in I}A_{i}\right)$.

      Por definición, esto significa que
      \begin{math}
        \exists\,x\in
        \bigcap_{i\in I}A_{i}\ni
        f\left(x\right)=
        y.
      \end{math}

      Esto es, $\exists\,x\ni\forall i\in I,x\in A_{i}$ y
      $f\left(x\right)=y$.

      Así, $\forall i\in I,y\in f\left(A_{i}\right)$.

      Por lo tanto, $y\in\bigcap_{i\in I}f\left(A_{i}\right)$.

      \part

      Sea
      \begin{math}
        f\colon
        \left\{1,2,3,4\right\}\to
        \left\{1,2,3,4\right\}.
      \end{math}
      $A=\left\{1,2,3\right\}$ y $B=\left\{2,3,4\right\}$.
      $A\cap B=\left\{2,3\right\}$.
      $f\left(1\right)=2$, $f\left(2\right)=3$, $f\left(3\right)=4$,
      $f\left(4\right)=2$.
      $f\left(A\cap B\right)=\left\{3,4\right\}$,
      $f\left(A\right)=\left\{2,3,4\right\}$,
      $f\left(B\right)=\left\{3,4,2\right\}$,
      \begin{math}
        f
        \left(A\right)\cap
        f\left(B\right)=
        \left\{2,3,4\right\}\neq
        \left\{3,4\right\}=
        f
        \left(A\cap B\right).
      \end{math}
    \end{parts}
  \end{solution}

  \question

  Sean $f\colon A\rightarrow B$, $g\colon B\rightarrow C$ funciones, pruebe que:

  \begin{parts}
    \part Si $f,g$ son inyectivas, entonces $g\circ f$ también lo es.

    \part Si $f,g$ son sobreyectivas, entonces $g\circ f$ también lo es.

    \part Si $f,g$ son biyectivas, entonces $g\circ f$ también lo es.
  \end{parts}

  \begin{solution}

    Sean $\mathcal{F}\colon A\to B$, $g\colon B\to C$.
    \begin{parts}
      \part

      Sean
      \begin{math}
        x_{1},x_{2}\in
        A\ni
        \left(g\circ f\right)
        \left(x_{1}\right)=
        \left(g\circ f\right)
        \left(x_{1}\right).
      \end{math}

      Entonces
      \begin{math}
        g\left(f\left(x_{1}\right)\right)=
        g\left(f\left(x_{2}\right)\right).
      \end{math}
      Así, $f\left(x_{1}\right)=f\left(x_{2}\right)$.

      Por lo tanto, $x_{1}=x_{2}$.
    \end{parts}
  \end{solution}
